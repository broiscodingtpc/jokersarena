import { useState } from "react";
import "./App.css";
import jokerImg from "./assets/joker.png"; // add your Sora Joker here
import coinImg from "./assets/coin.png"; // your $JOKER coin image

export default function App() {
  const [balance, setBalance] = useState(10000);
  const [bet, setBet] = useState(0);
  const [result, setResult] = useState(null);

  const handlePlay = () => {
    if (bet <= 0 || bet > balance) return;
    const didWin = Math.random() < 0.5;
    setBalance((prev) => prev + (didWin ? bet : -bet));
    setResult(didWin ? "win" : "lose");
  };

  const resetGame = () => {
    setBalance(10000);
    setResult(null);
    setBet(0);
  };

  return (
    <div className="arena">
      <h1>JOKER'S ARENA</h1>
      <img src={jokerImg} alt="Joker" className="joker" />

      <div className="balance">
        Mock Wallet: <strong>{balance} $JOKER</strong>
      </div>

      <div className="buttons">
        {[100, 500, 1000].map((amount) => (
          <button key={amount} onClick={() => setBet(amount)}>
            Bet {amount}
          </button>
        ))}
      </div>

      <button className="fight" onClick={handlePlay}>
        FIGHT THE JOKER
      </button>

      {result && (
        <div className={`result ${result}`}>
          {result === "win" ? "🎉 YOU WIN!" : "💀 JOKER WINS!"}
        </div>
      )}

      <button className="reset" onClick={resetGame}>
        Reset Game
      </button>
    </div>
  );
}
